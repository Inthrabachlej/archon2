# ARCHON 2.0 - AI CODE ORCHESTRATOR
## Technical Specification & Implementation Guide

---

## I. PROJECT GOAL

Build **Archon 2.0** - an autonomous system that:

1. Reads structured `deployment_plan.json` (user-created)
2. Uses **GPT-4o** (architect) to generate step-by-step instructions per module
3. Uses **Claude Sonnet 4.5** (builder) to generate code
4. Validates code (syntax, imports, dependencies)
5. Integrates modules (fixes imports, proper structure)
6. Deploys automatically (creates venv, installs deps, runs tests)
7. Outputs: **fully working, modular projects**

### Use Case:
```
Input:  deployment_plan.json (describes SaaS app: auth + payments + frontend)
Output: Working project, dependencies installed, ready to run
```

### Key Feature:
**Modular design** - backend/frontend/auth as reusable building blocks that can be:
- Generated separately
- Combined together
- Swapped out (different frontend, same backend)

## IV. EXECUTION FLOW
```
1. User creates deployment_plan.json
   ↓
2. orchestrator.build_project(plan_path)
   ↓
3. plan_parser: load + validate + sort by dependencies
   ↓
4. FOR EACH module (in dependency order):
   ├─ architect (GPT-4o): generate detailed instructions
   ├─ builder (Claude): generate code
   ├─ validator: check syntax + imports
   ├─ IF errors → retry with fixes (max 2 attempts)
   └─ Save files
   ↓
5. integrator: fix imports between modules
   ↓
6. deployer: create venv → install deps → run tests
   ↓
7. Return: Working project + build report
```

import json, os, shutil
from datetime import datetime

def _path(project, stage):
    return f"projects/{project}/state/{stage}.json"

def save_state(stage, data, project):
    os.makedirs(f"projects/{project}/state", exist_ok=True)
    with open(_path(project, stage), "w") as f:
        json.dump({"timestamp": datetime.now().isoformat(), "data": data}, f, indent=2)

def load_state(project, stage):
    with open(_path(project, stage)) as f:
        return json.load(f)

def resume_state(project, from_stage):
    data = load_state(project, from_stage)
    return {"status": "resumed", "from": from_stage, "data": data}

def rollback_state(project, to_stage):
    data = load_state(project, to_stage)
    save_state("rollback", data, project)
    return {"status": "rolled_back", "to": to_stage}

def clone_project(project):
    src = f"projects/{project}"
    dest = f"projects/{project}_clone"
    if os.path.exists(dest):
        shutil.rmtree(dest)
    shutil.copytree(src, dest)
    return {"status": "cloned", "clone_path": dest}

import os, json, re, requests
from core.state import save_state, load_state
from dotenv import load_dotenv
load_dotenv()

def extract_json_from_text(text: str):
    """Extract JSON from text that may contain markdown code blocks."""
    match = re.search(r'```json\s*(.*?)\s*```', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    match = re.search(r'```\s*(.*?)\s*```', text, re.DOTALL)
    if match:
        potential_json = match.group(1).strip()
        if potential_json.startswith('{'):
            return potential_json
    match = re.search(r'(\{.*\})', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    return text.strip()

def generate_code_for_folder(project: str, folder_path: str, description: str, blueprint: dict = None, completed_modules: list = None):
    """
    Generate code for a specific folder with full context awareness.
    
    Args:
        project: Project name
        folder_path: Path like 'backend/api' or 'frontend/src'
        description: What this module should do
        blueprint: Full blueprint dict from GPT-5 (provides context of other modules)
        completed_modules: List of already generated modules with their outputs
    """
    base = f"projects/{project}/output/{folder_path}"
    os.makedirs(base, exist_ok=True)

    # Build context-aware prompt
    context_info = ""
    
    if blueprint:
        context_info += f"\n\nFULL PROJECT BLUEPRINT:\n{json.dumps(blueprint, indent=2)}"
    
    if completed_modules:
        context_info += f"\n\nALREADY GENERATED MODULES:\n"
        for mod in completed_modules:
            context_info += f"- {mod['folder']}: {', '.join(mod.get('created', []))}\n"

    url = "https://api.anthropic.com/v1/messages"
    headers = {
        "x-api-key": os.getenv("CLOUD_API_KEY"),
        "anthropic-version": os.getenv("ANTHROPIC_VERSION", "2023-06-01"),
        "content-type": "application/json",
    }

    prompt = f"""You are a code generator for the folder '{folder_path}' in project '{project}'.

TASK: {description}
{context_info}

Generate ONLY the code files needed for THIS folder ('{folder_path}').
Return valid JSON mapping filenames to their complete code content.

CRITICAL RULES:
1. Return ONLY JSON, no explanations
2. Use relative imports when referencing other modules from the blueprint
3. File paths should be relative to '{folder_path}' (just filename, no subdirectories)
4. Each file must have complete, working code
5. If this module depends on others, import them correctly

JSON FORMAT:
{{
  "main.py": "from fastapi import FastAPI\\napp = FastAPI()\\n...",
  "routes.py": "from fastapi import APIRouter\\nrouter = APIRouter()\\n...",
  "models.py": "from pydantic import BaseModel\\n..."
}}

Return ONLY the JSON object.
"""

    payload = {
        "model": os.getenv("CLOUD_MODEL", "claude-sonnet-4-5-20250929"),
        "max_tokens": 4096,
        "messages": [
            {"role": "user", "content": [{"type": "text", "text": prompt}]}
        ],
    }

    try:
        resp = requests.post(url, headers=headers, json=payload, timeout=180)
        resp.raise_for_status()
        raw = resp.json()
        
        text = ""
        if isinstance(raw.get("content"), list):
            for part in raw["content"]:
                if part.get("type") == "text":
                    text += part.get("text", "")

        clean = extract_json_from_text(text)
        files = json.loads(clean)
        
        if not isinstance(files, dict):
            raise ValueError(f"Expected dict, got {type(files)}")
            
    except requests.exceptions.Timeout:
        files = {"error": "Claude timeout after 180s"}
    except json.JSONDecodeError as e:
        files = {"error": f"JSON parse failed: {e}", "raw_response": text[:500]}
    except Exception as e:
        files = {"error": str(e)}

    # Write files to disk
    created_files = []
    for filename, content in files.items():
        if filename == "error":
            continue
        path = os.path.join(base, filename)
        os.makedirs(os.path.dirname(path), exist_ok=True)
        with open(path, "w") as f:
            f.write(str(content))
        created_files.append(path)

    report = {
        "folder": folder_path,
        "created": created_files,
        "files": files if "error" not in files else {"error": files["error"]}
    }
    save_state(f"worker_{folder_path.replace('/', '_')}", report, project)
    return report

def extract_json_from_text(text: str):
    """Extract JSON from text that may contain markdown code blocks or explanations."""
    # Try to find JSON in markdown code block
    match = re.search(r'```json\s*(.*?)\s*```', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    
    # Try to find JSON in generic code block
    match = re.search(r'```\s*(.*?)\s*```', text, re.DOTALL)
    if match:
        potential_json = match.group(1).strip()
        if potential_json.startswith('{') or potential_json.startswith('['):
            return potential_json
    
    # Try to find raw JSON (starts with { or [)
    match = re.search(r'(\{.*\}|\[.*\])', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    
    # Last resort: return as-is and let json.loads fail with clear error
    return text.strip()

def generate_blueprint(project: str, description: str):
    url = "https://api.openai.com/v1/chat/completions"
    headers = {
        "Authorization": f"Bearer {os.getenv('OPENAI_API_KEY')}",
        "Content-Type": "application/json"
    }
    prompt = f"""
Decompose the following software concept into distinct modules with clear sub-tasks:
'{description}'.

Return ONLY valid JSON (no explanations, no markdown) in this exact format:
{{
  "backend": "FastAPI with endpoints /auth, /reels, /payments",
  "frontend": "React + Tailwind login + dashboard",
  "database": "SQLite or Postgres setup",
  "auth": "JWT token login and signup"
}}

CRITICAL: Return ONLY the JSON object, nothing else.
"""

    payload = {
        "model": os.getenv("OPENAI_MODEL", "gpt-4o"),
        "messages": [{"role": "user", "content": prompt}],
        "temperature": 0.7
    }

    try:
        resp = requests.post(url, headers=headers, json=payload, timeout=90)
        resp.raise_for_status()
        data = resp.json()
        
        # Handle different response formats
        if "choices" in data:
            blueprint_raw = data["choices"][0]["message"]["content"]
        elif "output" in data:
            blueprint_raw = data["output"]["content"][0]["text"]
        else:
            raise ValueError(f"Unexpected GPT response format: {data}")
        
        # Extract JSON from potentially wrapped text
        json_str = extract_json_from_text(blueprint_raw)
        blueprint = json.loads(json_str)
        
        # Validate it's a dict
        if not isinstance(blueprint, dict):
            raise ValueError(f"Blueprint must be a dict, got {type(blueprint)}")
        
        save_state("blueprint", blueprint, project)
        return blueprint
        
    except requests.exceptions.Timeout:
        raise Exception(f"GPT-5 timeout after 90s. Try shorter description or split into modules.")
    except json.JSONDecodeError as e:
        raise Exception(f"Failed to parse GPT-5 response as JSON: {e}\nRaw response: {blueprint_raw[:500]}")
    except Exception as e:
        raise Exception(f"Blueprint generation failed: {e}")

def create_structure(project: str, layout: dict):
    """
    layout: dict typu
    {
      "backend": {
         "api": ["main.py", "routes.py"],
         "models": ["user.py"]
      },
      "frontend": {
         "src": ["index.html", "app.js"]
      },
      "shared": ["README.md"]
    }
    """
    base_dir = f"projects/{project}/output"
    created = []

    def make_dir(path):
        if not os.path.exists(path):
            os.makedirs(path, exist_ok=True)

    for folder, content in layout.items():
        folder_path = os.path.join(base_dir, folder)
        make_dir(folder_path)

        if isinstance(content, dict):
            for subfolder, files in content.items():
                sub_path = os.path.join(folder_path, subfolder)
                make_dir(sub_path)
                for f in files:
                    full_path = os.path.join(sub_path, f)
                    open(full_path, "w").close()
                    created.append(full_path)
        elif isinstance(content, list):
            for f in content:
                full_path = os.path.join(folder_path, f)
                open(full_path, "w").close()
                created.append(full_path)
        else:
            full_path = os.path.join(base_dir, folder)
            open(full_path, "w").close()
            created.append(full_path)

    save_state("stage_structure", {"created": created}, project)
    return {"created": created}

def extract_json_from_text(text: str):
    """Extract JSON from markdown or plain text."""
    match = re.search(r'```json\s*(.*?)\s*```', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    match = re.search(r'```\s*(.*?)\s*```', text, re.DOTALL)
    if match:
        potential = match.group(1).strip()
        if potential.startswith('{'):
            return potential
    match = re.search(r'(\{.*\})', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    return text.strip()

import os, json
from core.state import save_state, load_state
from core.folder_worker import generate_code_for_folder
from core.structure import create_structure

def validate_dependencies(modules: list):
    """Ensure all dependencies exist and no circular deps."""
    module_ids = {m["id"] for m in modules}
    
    for module in modules:
        for dep in module.get("dependencies", []):
            if dep not in module_ids:
                raise ValueError(f"Module {module['id']} depends on unknown module: {dep}")
    
    return True

def sort_modules_by_dependencies(modules: list):
    """Topological sort: modules with no deps first, then their dependents."""
    sorted_modules = []
    processed = set()
    
    # Keep trying until all modules are processed
    while len(sorted_modules) < len(modules):
        made_progress = False
        
        for module in modules:
            if module["id"] in processed:
                continue
            
            # Check if all dependencies are processed
            deps = module.get("dependencies", [])
            if all(dep in processed for dep in deps):
                sorted_modules.append(module)
                processed.add(module["id"])
                made_progress = True
        
        if not made_progress:
            raise ValueError("Circular dependency detected in plan")
    
    return sorted_modules

def execute_deployment_plan(project: str):
    """
    Execute deployment plan step by step with context.
    
    1. Load plan from state
    2. Sort modules by dependencies
    3. Execute each module in order
    4. Pass accumulated context to each step
    5. Validate after each step
    """
    base_dir = f"projects/{project}"
    
    # Load plan
    try:
        plan_state = load_state(project, "deployment_plan")
        plan = plan_state["data"]
    except Exception as e:
        raise Exception(f"Failed to load deployment plan: {e}")
    
    modules = plan.get("modules", [])
    if not modules:
        raise ValueError("Plan has no modules")
    
    # Validate dependencies
    validate_dependencies(modules)
    
    # Sort by dependencies
    sorted_modules = sort_modules_by_dependencies(modules)
    
    print(f"[PLAN EXECUTOR] Executing {len(sorted_modules)} modules in order:")
    for i, mod in enumerate(sorted_modules, 1):
        deps = mod.get("dependencies", [])
        dep_str = f" (depends on: {', '.join(deps)})" if deps else " (no dependencies)"
        print(f"  {i}. {mod['name']}{dep_str}")
    
    # Create folder structure
    layout = {mod["name"]: [] for mod in sorted_modules}
    create_structure(project, layout)
    
    # Execute modules in order with context
    results = []
    completed_modules = []
    
    for idx, module in enumerate(sorted_modules, 1):
        module_id = module["id"]
        module_name = module["name"]
        module_desc = module["description"]
        
        print(f"\n[PLAN EXECUTOR] → [{idx}/{len(sorted_modules)}] Executing: {module_name}")
        print(f"[PLAN EXECUTOR]   Description: {module_desc}")
        
        # Build context from dependencies
        dependency_context = []
        for dep_id in module.get("dependencies", []):
            # Find completed module with this id
            for completed in completed_modules:
                if completed.get("module_id") == dep_id:
                    dependency_context.append(completed)
                    break
        
        # Generate code with full context
        result = generate_code_for_folder(
            project=project,
            folder_path=module_name,
            description=module_desc,
            blueprint=plan,  # Full plan as context
            completed_modules=dependency_context  # Only direct dependencies
        )
        
        # Add module_id to result for dependency tracking
        result["module_id"] = module_id
        result["module_name"] = module_name
        
        results.append(result)
        completed_modules.append(result)
        
        # Check for errors
        if "error" in result.get("files", {}):
            print(f"[PLAN EXECUTOR] ⚠ Warning: {module_name} generated with errors: {result['files']['error']}")
        else:
            file_count = len(result.get("created", []))
            print(f"[PLAN EXECUTOR] ✓ {module_name} complete: {file_count} files")
    
    # Save execution results
    execution_report = {
        "plan": plan,
        "execution_order": [m["id"] for m in sorted_modules],
        "results": results,
        "total_files": sum(len(r.get("created", [])) for r in results)
    }
    save_state("plan_execution", execution_report, project)
    
    print(f"\n[PLAN EXECUTOR] ✓ Execution complete: {execution_report['total_files']} files generated")
    
    return execution_report
